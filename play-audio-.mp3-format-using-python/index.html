第二种方式pygame.mixer.music(推荐使用)

from ast import While
import pygame

"""初始化pygame"""
pygame.mixer.init()

"""加载音乐文件"""
pygame.mixer.music.load('your music file path')

"""设置音量"""
pygame.mixer.music.set_volume(0.2)

"""播放音乐"""
pygame.mixer.music.play()

"""播放完成再结束"""
while pygame.mixer.music.get_busy():
    pass
print('播放结束')
